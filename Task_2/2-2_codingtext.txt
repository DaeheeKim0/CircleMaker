#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> board, int k) {
    int answer = 0;
    int height = board.size();
    int width = board.front().size();
    
    for (int i=0; i<height; i++) {
        if (i > k) {
            break;
        }
        for (int j=0; j<width; j++) {
            if (i + j > k) {
                break;
            }
            answer += board[i][j];
        }
    }
    return answer;
}
